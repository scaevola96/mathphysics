\begin{equation}
\begin{aligned}
    \int_{0}^{+\infty} x^n e^{-x^r} \diff x = \left| x^r = t \right| = \frac{1}{r} \int_{0}^{+\infty} t^{1+\frac{n-r}{r}} e^{-t} t^{\frac{1}{r} -1} \diff t
    =  \frac{1}{r} \int_{0}^{+\infty} t^{\frac{n+1}{r}-1} e^{-t} \diff t =  \frac{1}{r} \Gamma\left({\frac{n+1}{r}}\right)
\end{aligned}
\end{equation}
